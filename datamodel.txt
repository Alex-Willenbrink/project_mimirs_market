
SEQUELIZE COMMANDS FOR SETTING UP MIGRATIONS AND MODELS
User:
  sequelize model:create --name User --attributes "username:string email:string isAdmin:boolean"

Product:
  sequelize model:create --name Product --attributes "name:string picture:integer price:integer stock:integer sku:string description:string"

Category:
  sequelize model:create --name Category --attributes "name:string"

Order
  sequelize model:create --name Order --attributes "shippedStatus:boolean shipState:integer shipCountry:string shipCity:string price:float date:date userId:integer"

State
  sequelize model:create --name State --attributes "abbr:string"

ProductsOrder
  sequelize model:create --name ProductsOrder --attributes "productId:integer orderId:integer"

ProductsCategory:
  sequelize model:create --name ProductsCategory --attributes "productId:integer categoryId:integer"
